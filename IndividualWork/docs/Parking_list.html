def self.intialize_DB(entity)
        case entity
                when :owner
                        Parking_list.new(Owners_list_DB.new())
                when :guard
                        Parking_list.new(Guards_list_DB.new())
                when :auto
                        Parking_list.new(Auto_list_DB.new())
                else
                        raise ArgumentError, "Invalid argument #{entity}"
        end         
end
